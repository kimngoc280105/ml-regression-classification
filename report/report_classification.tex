\section{Giới thiệu}
\label{sec:intro}

Báo cáo này trình bày kết quả thực nghiệm của \textbf{Phần 2 -- Classification} trong Đồ Án 1 môn CSC14005 (Machine Learning).
Mục tiêu là xây dựng và so sánh các mô hình phân loại \emph{từ đầu} (from scratch, không dùng \texttt{sklearn} cho phần huấn luyện) nhằm phân loại thu nhập của người theo hai lớp: $\leq$50K và $>$50K.

\subsection{Bài toán}
\begin{itemize}
  \item \textbf{Dữ liệu:} Adult Census Income Dataset (UCI Repository), được lưu tại \texttt{data/adult.csv}.
  \item \textbf{Mục tiêu:} Dự đoán nhãn nhị phân \texttt{income} $\in \{0, 1\}$ ($0 \leftrightarrow {\leq}50K$, $1 \leftrightarrow {>}50K$).
  \item \textbf{Input:} 14 đặc trưng gồm tuổi, trình độ học vấn, nghề nghiệp, số giờ làm việc, \ldots
\end{itemize}

\subsection{Các mô hình đã triển khai}
Dự án triển khai 13 mô hình phân loại:
\begin{enumerate}
  \item Logistic Regression -- Gradient Descent (GD)
  \item Logistic Regression -- Newton-Raphson / IRLS
  \item Softmax Regression (Multi-class)
  \item One-vs-Rest (OvR)
  \item One-vs-One (OvO)
  \item Linear Discriminant Analysis (LDA)
  \item Quadratic Discriminant Analysis (QDA)
  \item Perceptron
  \item Regularized Logistic Regression (L1 / L2)
  \item Probit Regression
  \item Logistic Regression + Laplace Approximation
  \item Kernel Logistic Regression (RBF)
  \item Gaussian Naive Bayes
\end{enumerate}

\section{Tiền xử lý dữ liệu}
\label{sec:preprocessing}

\subsection{Xử lý giá trị thiếu}
Bộ dữ liệu Adult chứa ký tự \texttt{?} thay cho giá trị thiếu trong các cột phân loại (\texttt{workclass}, \texttt{occupation}, \texttt{native-country}). Chiến lược xử lý:
\begin{itemize}
  \item Thay thế \texttt{?} bằng giá trị mode của cột tương ứng (imputation bằng most-frequent).
\end{itemize}

\subsection{Mã hóa đặc trưng phân loại}
Các đặc trưng phân loại (categorical) được mã hóa bằng \emph{Label Encoding}:
$$\text{LabelEncoder}: \texttt{workclass}, \texttt{education}, \texttt{marital-status}, \ldots \to \mathbb{Z}$$

\subsection{Chuẩn hóa đặc trưng số}
Các đặc trưng số được chuẩn hóa bằng \emph{Standardization}:
$$x' = \frac{x - \mu}{\sigma}$$
Trong đó $\mu$ và $\sigma$ được tính \textbf{chỉ trên tập train} để tránh \emph{data leakage}.

\subsection{Phân chia dữ liệu}
\begin{itemize}
  \item Train / Validation / Test: $70\%$ / $10\%$ / $20\%$ (stratified theo nhãn để giữ nguyên tỷ lệ lớp).
  \item Hàm: \texttt{train\_test\_split} với \texttt{random\_state=42}, \texttt{stratify=y}.
\end{itemize}

Cụ thể, với tổng $N$ mẫu:
$$N_{\text{train}} = 0.70N, \quad N_{\text{val}} = 0.10N, \quad N_{\text{test}} = 0.20N$$

\section{Các mô hình phân loại}
\label{sec:models}

\subsection{Logistic Regression -- Gradient Descent}
\label{ssec:lr_gd}

\subsubsection{Nền tảng lý thuyết}

Logistic Regression mô hình hóa xác suất có điều kiện:
$$P(y=1 \mid \mathbf{x}) = \sigma(\mathbf{\theta}^\top \mathbf{x}) = \frac{1}{1 + e^{-\mathbf{\theta}^\top \mathbf{x}}}$$
trong đó $\sigma(\cdot)$ là hàm sigmoid.

\textbf{Hàm mất mát (Binary Cross-Entropy):}
$$\mathcal{L}(\mathbf{\theta}) = -\frac{1}{m}\sum_{i=1}^{m} \left[y^{(i)}\log \hat{y}^{(i)} + (1-y^{(i)})\log(1-\hat{y}^{(i)})\right]$$

\textbf{Cập nhật tham số (Gradient Descent):}
$$\mathbf{\theta} \leftarrow \mathbf{\theta} - \alpha \nabla_\theta \mathcal{L}(\mathbf{\theta})$$
với gradient:
$$\nabla_\theta \mathcal{L} = \frac{1}{m} \mathbf{X}^\top (\hat{\mathbf{y}} - \mathbf{y})$$

\textbf{Regularization:}
\begin{align*}
  \text{L2 (Ridge):} \quad & \mathcal{L}_{L2} = \mathcal{L} + \frac{\lambda}{2m}\|\mathbf{\theta}_{1:}\|^2 \\
  \text{L1 (Lasso):} \quad & \mathcal{L}_{L1} = \mathcal{L} + \frac{\lambda}{m}\|\mathbf{\theta}_{1:}\|_1
\end{align*}

\subsection{Logistic Regression -- Newton-Raphson / IRLS}
\label{ssec:lr_newton}

\subsubsection{Nền tảng lý thuyết}

Newton-Raphson sử dụng Hessian để hội tụ nhanh hơn GD:
$$\mathbf{\theta} \leftarrow \mathbf{\theta} - \mathbf{H}^{-1} \nabla_\theta \mathcal{L}$$

\textbf{Hessian:}
$$\mathbf{H} = \frac{1}{m}\mathbf{X}^\top \mathbf{R} \mathbf{X}$$
với $\mathbf{R} = \text{diag}(\hat{y}^{(i)}(1-\hat{y}^{(i)}))$.

\subsubsection{So sánh GD vs Newton-Raphson}

\begin{figure}[H]
  \centering
  \includegraphics[width=0.75\textwidth]{figures/gd_vs_newton.png}
  \caption{Hội tụ Loss theo epochs và wall-clock time: GD vs Newton-Raphson.}
  \label{fig:gd_newton}
\end{figure}

Nhận xét: Newton-Raphson hội tụ chỉ sau $\sim$8 iterations so với 300 iterations của GD, nhưng mỗi iteration tốn chi phí $O(nd^2)$ để tính Hessian.

\subsection{Softmax Regression (Multi-class)}
\label{ssec:softmax}

\subsubsection{Nền tảng lý thuyết}

Softmax Regression tổng quát hóa Logistic Regression sang $K$ lớp. Xác suất lớp $k$ được tính bằng hàm softmax:
$$P(y=k \mid \mathbf{x}) = \frac{e^{\boldsymbol{\theta}_k^\top \mathbf{x}}}{\sum_{j=1}^{K} e^{\boldsymbol{\theta}_j^\top \mathbf{x}}}$$

\textbf{Hàm mất mát (Multi-class Cross-Entropy):}
$$\mathcal{L}(\boldsymbol{\Theta}) = -\frac{1}{m}\sum_{i=1}^{m}\sum_{k=1}^{K} \mathbf{1}[y^{(i)}=k]\log P(y^{(i)}=k \mid \mathbf{x}^{(i)})$$

\textbf{Gradient (cập nhật ma trận trọng số $\boldsymbol{\Theta} \in \mathbb{R}^{d \times K}$):}
$$\frac{\partial \mathcal{L}}{\partial \boldsymbol{\Theta}} = \frac{1}{m}\mathbf{X}^\top(\hat{\mathbf{P}} - \mathbf{Y}_{\text{onehot}})$$

Trong đó $\hat{\mathbf{P}} \in \mathbb{R}^{m \times K}$ là ma trận xác suất dự đoán và $\mathbf{Y}_{\text{onehot}}$ là nhãn one-hot.

\subsubsection{Quan hệ với Logistic Regression}
Khi $K=2$, Softmax Regression tương đương Logistic Regression nhị phân. Với Adult dataset (bài toán nhị phân), Softmax được dùng như một kiểm chứng sự nhất quán.

\subsection{One-vs-Rest (OvR)}
\label{ssec:ovr}

\subsubsection{Nền tảng lý thuyết}

OvR (hay One-vs-All, OvA) huấn luyện $K$ bộ phân loại nhị phân, mỗi bộ phân biệt lớp $k$ với tất cả các lớp còn lại:
$$f_k(\mathbf{x}) = P(y=k \mid \mathbf{x}) \approx \sigma(\boldsymbol{\theta}_k^\top \mathbf{x})$$

\textbf{Quy tắc quyết định:}
$$\hat{y} = \arg\max_{k \in \{1,\ldots,K\}} f_k(\mathbf{x})$$

\textbf{Triển khai:} Mỗi $f_k$ là một \texttt{LogisticRegressionGD} độc lập, huấn luyện trên nhãn nhị phân $y_k^{(i)} = \mathbf{1}[y^{(i)} = k]$.

\subsubsection{Phân tích}
\begin{itemize}
  \item \textbf{Ưu điểm:} Đơn giản, song song hóa được; tổng chi phí huấn luyện $K$ lần so với một bộ phân loại nhị phân.
  \item \textbf{Hạn chế:} Lớp mất cân bằng nghiêm trọng hơn (mỗi bộ phân loại $f_k$ có $\sim(K-1)/K$ mẫu âm).
  \item \textbf{Độ phức tạp:} $O(K \cdot T \cdot nd)$ với $T$ iterations, $n$ mẫu, $d$ đặc trưng.
\end{itemize}

% -----------------------------------------------------------

\subsection{One-vs-One (OvO)}
\label{ssec:ovo}

\subsubsection{Nền tảng lý thuyết}

OvO huấn luyện $\binom{K}{2}$ bộ phân loại nhị phân, mỗi bộ phân biệt một cặp lớp $(i, j)$:
$$f_{ij}(\mathbf{x}) = \sigma(\boldsymbol{\theta}_{ij}^\top \mathbf{x}) \in [0,1]$$

\textbf{Quyết định bằng biểu quyết đa số (Majority Voting):}
$$\hat{y} = \arg\max_{k} \text{votes}(k), \quad \text{votes}(k) = \sum_{j \neq k} \mathbf{1}[f_{kj}(\mathbf{x}) > 0.5]$$

\subsubsection{So sánh OvR vs OvO}
\begin{table}[H]
  \centering
  \caption{So sánh chiến lược OvR và OvO ($K$ lớp)}
  \begin{tabular}{lcc}
    \toprule
    \textbf{Tiêu chí} & \textbf{OvR} & \textbf{OvO} \\
    \midrule
    Số bộ phân loại & $K$ & $\binom{K}{2} = K(K-1)/2$ \\
    Kích thước tập train mỗi bộ & $n$ (toàn bộ) & $\sim 2n/K$ (chỉ 2 lớp) \\
    Chi phí huấn luyện & $O(K \cdot nd)$ & $O(K^2/2 \cdot n/K \cdot d)$ \\
    Thiên vị mất cân bằng & Cao hơn & Thấp hơn \\
    \bottomrule
  \end{tabular}
\end{table}

\textit{Ghi chú:} Với bài toán nhị phân ($K=2$), OvR và OvO đều tương đương Logistic Regression thông thường.

\subsection{Linear Discriminant Analysis (LDA)}
\label{ssec:lda}

\subsubsection{Nền tảng lý thuyết}

LDA là mô hình \emph{generative} giả định mỗi lớp có phân phối Gaussian với \textbf{ma trận hiệp phương sai chung}:
$$p(\mathbf{x} \mid y=k) = \mathcal{N}(\mathbf{x}; \boldsymbol{\mu}_k, \boldsymbol{\Sigma})$$

\textbf{Hàm phân biệt tuyến tính:}
$$\delta_k(\mathbf{x}) = \mathbf{x}^\top \boldsymbol{\Sigma}^{-1} \boldsymbol{\mu}_k - \frac{1}{2}\boldsymbol{\mu}_k^\top \boldsymbol{\Sigma}^{-1} \boldsymbol{\mu}_k + \log\pi_k$$

\textbf{Shared covariance:}
$$\boldsymbol{\Sigma} = \frac{1}{n - K}\sum_{k=1}^{K}\sum_{i: y_i=k} (\mathbf{x}_i - \boldsymbol{\mu}_k)(\mathbf{x}_i - \boldsymbol{\mu}_k)^\top$$

\textbf{Fisher Discriminant Ratio} đo mức độ phân biệt của từng đặc trưng:
$$\text{FDR}_j = \frac{S_B^{(j)}}{S_W^{(j)}}$$

% -----------------------------------------------------------

\subsection{Quadratic Discriminant Analysis (QDA)}
\label{ssec:qda}

QDA cho phép mỗi lớp có ma trận hiệp phương sai \textbf{riêng}:
$$p(\mathbf{x} \mid y=k) = \mathcal{N}(\mathbf{x}; \boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k)$$

\textbf{Hàm phân biệt bậc hai:}
$$\delta_k(\mathbf{x}) = -\frac{1}{2}\log|\boldsymbol{\Sigma}_k| - \frac{1}{2}(\mathbf{x}-\boldsymbol{\mu}_k)^\top \boldsymbol{\Sigma}_k^{-1}(\mathbf{x}-\boldsymbol{\mu}_k) + \log\pi_k$$

\begin{table}[H]
  \centering
  \caption{So sánh số tham số: LDA vs QDA ($K=2$, $d=$ số đặc trưng)}
  \begin{tabular}{lcc}
    \toprule
    & \textbf{LDA} & \textbf{QDA} \\
    \midrule
    Mean vectors & $K \cdot d$ & $K \cdot d$ \\
    Covariance matrices & $\frac{d(d+1)}{2}$ (shared) & $K \cdot \frac{d(d+1)}{2}$ (per-class) \\
    Priors & $K-1$ & $K-1$ \\
    \midrule
    \textbf{Tổng} & $\sim Kd + d^2/2$ & $\sim Kd + Kd^2/2$ \\
    \bottomrule
  \end{tabular}
\end{table}

\subsection{Gaussian Naive Bayes}
\label{ssec:gnb}

Naive Bayes giả định các đặc trưng \textbf{độc lập có điều kiện} (diagonal covariance):
$$p(\mathbf{x} \mid y=k) = \prod_{j=1}^{d} \mathcal{N}(x_j; \mu_{kj}, \sigma_{kj}^2)$$

Log-likelihood:
$$\log P(y=k \mid \mathbf{x}) \propto \sum_{j=1}^{d} \left[-\frac{(x_j - \mu_{kj})^2}{2\sigma_{kj}^2} - \frac{1}{2}\log(2\pi\sigma_{kj}^2)\right] + \log\pi_k$$

% -----------------------------------------------------------

\subsection{Perceptron}
\label{ssec:perceptron}

Cập nhật Perceptron (nhãn $\in \{-1, +1\}$):
$$\mathbf{w} \leftarrow \mathbf{w} + \alpha \cdot y_i \cdot \mathbf{x}_i \quad \text{nếu } y_i \cdot (\mathbf{w}^\top \mathbf{x}_i) \leq 0$$

% -----------------------------------------------------------

\subsection{Probit Regression}
\label{ssec:probit}

Probit Regression sử dụng hàm CDF Gaussian thay cho sigmoid:
$$P(y=1 \mid \mathbf{x}) = \Phi(\mathbf{\theta}^\top \mathbf{x}) = \int_{-\infty}^{\mathbf{\theta}^\top \mathbf{x}} \frac{1}{\sqrt{2\pi}}e^{-t^2/2}dt$$

Gradient:
$$\nabla_\theta \mathcal{L} = \frac{1}{m}\mathbf{X}^\top \left[\frac{\hat{y} - y}{\hat{y}(1-\hat{y})} \cdot \phi(\mathbf{\theta}^\top \mathbf{x})\right]$$

% -----------------------------------------------------------

\subsection{Kernel Logistic Regression (RBF)}
\label{ssec:kernel_lr}

Sử dụng biểu diễn dual với kernel RBF:
$$K(\mathbf{x}_i, \mathbf{x}_j) = \exp\left(-\gamma\|\mathbf{x}_i - \mathbf{x}_j\|^2\right)$$

Dự đoán: $f(\mathbf{x}) = \sum_i \alpha_i K(\mathbf{x}_i, \mathbf{x})$

% -----------------------------------------------------------

\subsection{Logistic Regression -- Laplace Approximation}
\label{ssec:laplace}

Laplace Approximation xấp xỉ posterior bằng Gaussian tại ước lượng MAP:
$$q(\mathbf{\theta}) = \mathcal{N}(\mathbf{\theta}; \mathbf{\theta}_{MAP}, \mathbf{H}^{-1})$$

Dự đoán với bất định (uncertainty):
$$P(y=1 \mid \mathbf{x}, \mathcal{D}) \approx \sigma\!\left(\frac{\mathbf{\theta}_{MAP}^\top \mathbf{x}}{\sqrt{1 + \frac{\pi}{8}\mathbf{x}^\top\mathbf{H}^{-1}\mathbf{x}}}\right)$$

\section{Phương pháp đánh giá}
\label{sec:evaluation}

Do dữ liệu \emph{imbalanced} ($\approx$3:1), phương pháp đánh giá chính:

\begin{itemize}
  \item \textbf{Accuracy}: $\frac{TP+TN}{TP+TN+FP+FN}$ (tham khảo, thiên vị với lớp đa số)
  \item \textbf{Precision}: $\frac{TP}{TP+FP}$
  \item \textbf{Recall}: $\frac{TP}{TP+FN}$ (quan trọng -- tránh bỏ sót lớp $>$50K)
  \item \textbf{F1-Score}: $\frac{2\cdot\text{Precision}\cdot\text{Recall}}{\text{Precision}+\text{Recall}}$ (\textbf{metric chính})
  \item \textbf{AUC-ROC}: Diện tích dưới đường ROC
  \item \textbf{McNemar Test}: Kiểm định thống kê chênh lệch giữa hai mô hình
\end{itemize}

\section{Thảo luận}

\subsection{Mô hình tốt nhất}

Dựa trên kết quả thực nghiệm tổng hợp từ Bảng~\ref{tab:results} và Bảng~\ref{tab:cv},
mô hình \textbf{Logistic Regression Newton-Raphson (LR-Newton)} được đánh giá là tốt nhất
cho bài toán phân loại thu nhập trên bộ dữ liệu Adult Census Income.

\begin{itemize}
  \item \textbf{Accuracy}: $0.8267$ -- cao nhất trong nhóm các mô hình không dùng class weighting.
  \item \textbf{F1-Score (class $>$50K)}: $0.5529$ -- cao nhất trong nhóm mô hình cân bằng Precision/Recall.
  \item \textbf{AUC}: $0.8468$ -- phản ánh khả năng phân tách tốt giữa hai lớp.
  \item \textbf{5-Fold CV}: F1 $= 0.5568 \pm 0.0088$, Accuracy $= 0.8272 \pm 0.0027$ -- ổn định,
        std nhỏ xác nhận mô hình không overfit.
  \item \textbf{McNemar Test} (LR-GD vs.\ LR-Newton): $\chi^2 = 4.82$, $p = 0.028$ --
        LR-Newton thực sự phân lớp đúng \emph{thêm} một số mẫu so với LR-GD, sự khác biệt có ý nghĩa thống kê.
\end{itemize}

\noindent\textbf{Lý do LR-Newton tốt hơn LR-GD:}
Newton-Raphson sử dụng đạo hàm bậc hai (Hessian) để tính bước nhảy tối ưu, nên hội tụ
trong $\sim$8 iterations thay vì 300 iterations của Gradient Descent.
Trên bộ dữ liệu Adult ($\sim$32K mẫu, 12 features), chi phí tính Hessian $O(D^2)$ vẫn
chấp nhận được và đổi lại có nghiệm chất lượng cao hơn.

\noindent\textbf{Lưu ý về LR-Weighted:} Nếu ưu tiên \textbf{Recall lớp thiểu số} ($>$50K),
mô hình \textbf{Logistic Regression với class weighting} đạt Recall $= 0.7666$ (cao nhất),
nhưng Precision giảm mạnh ($0.5053$) và Accuracy chỉ $0.7631$.
Đây là sự đánh đổi (trade-off) giữa Recall và Precision trong bài toán mất cân bằng lớp.

\subsection{Ảnh hưởng của imbalanced data}

Bộ dữ liệu Adult có mất cân bằng lớp đáng kể: \textbf{75.93\% mẫu thuộc lớp $\leq$50K}
và chỉ \textbf{24.07\% thuộc lớp $>$50K}. Điều này ảnh hưởng trực tiếp đến hiệu năng:

\begin{itemize}
  \item \textbf{Không dùng class weighting}: Tất cả mô hình thiên về lớp đa số.
        Recall lớp $>$50K dao động trong khoảng $0.30$--$0.45$, phản ánh mô hình bỏ sót
        phần lớn người có thu nhập cao.
        Phân tích lỗi cho thấy \textbf{78.4\% lỗi} là ``$>$50K bị dự đoán nhầm thành $\leq$50K''.
  \item \textbf{Dùng class weighting (LR-Weighted)}: Recall lớp $>$50K tăng lên $0.7666$
        (cải thiện $\approx$+32 điểm phần trăm so với LR-GD $0.4247$), nhưng Precision
        giảm từ $0.73$ xuống $0.51$ và Accuracy giảm từ $0.82$ xuống $0.76$.
  \item \textbf{Kết luận trade-off}: Với bài toán phân loại thu nhập, nếu chi phí bỏ sót
        người có thu nhập cao ($>$50K) lớn hơn chi phí cảnh báo nhầm (ví dụ: hệ thống
        cấp tín dụng), nên dùng LR-Weighted. Ngược lại, nếu cần cân bằng Precision/Recall,
        LR-Newton là lựa chọn tốt hơn.
\end{itemize}

\section{Kết luận}
\label{sec:conclusion}

Báo cáo đã:
\begin{enumerate}
  \item Phân tích EDA toàn diện bộ dữ liệu Adult Census Income (32{,}561 mẫu, 14 features):
        xử lý missing values, phát hiện outlier (đặc biệt \texttt{hours-per-week} có
        27.7\% outlier theo IQR), mã hóa categorical features.
  \item Triển khai \textbf{10 mô hình phân loại} \emph{from scratch} bằng NumPy:
        LR-GD, LR-Newton, LDA, QDA, Probit, Laplace, Reg.\ LR L1/L2, LR-Weighted, GNB.
  \item So sánh hiệu năng trên tập test (Accuracy, Precision, Recall, F1, AUC, AP)
        và xác nhận tính ổn định qua \textbf{5-Fold Cross-Validation}.
  \item Kiểm định thống kê bằng \textbf{McNemar Test} cho 4 cặp mô hình.
  \item Phân tích tác động của \textbf{imbalanced data} và kỹ thuật class weighting.
\end{enumerate}

\textbf{Kết luận chính:}
\begin{itemize}
  \item \textbf{LR-Newton} là mô hình tốt nhất cho binary classification trên Adult:
        F1 $= 0.5529$, AUC $= 0.8468$, hội tụ trong $\approx 8$ iterations.
  \item \textbf{Newton-Raphson hội tụ nhanh hơn Gradient Descent $\sim$37 lần}
        về số iterations (8 vs.\ 300), xác nhận bởi McNemar Test ($p = 0.028$).
  \item \textbf{LDA vượt QDA} trên dữ liệu này vì dù QDA linh hoạt hơn, tập Adult có
        imbalance nghiêm trọng khiến QDA ước lượng covariance lớp thiểu số kém chính xác hơn,
        dẫn đến Recall $= 0.30$ (thấp hơn LDA $0.39$).
  \item \textbf{Class weighting} cải thiện Recall lớp thiểu số từ $0.43$ lên $0.77$
        (+34 điểm phần trăm), nhưng đánh đổi Precision giảm từ $0.73$ xuống $0.51$.
  \item \textbf{Regularization L2} ($\lambda = 0.001$) cho kết quả giống hệt LR-GD
        ($\chi^2 = 0$, $p = 1$) -- cần tăng $\lambda$ để thấy hiệu ứng regularization rõ hơn.
\end{itemize}

regularization (L1/L2), và tác động của class weighting.

\noindent\textbf{Nhận xét:}

ít có nguy cơ overfit, regularization ổn định nhưng không mang lại cải thiện đáng
kể. L1 với $\lambda=1.0$ mới bắt đầu ảnh hưởng nhẹ đến F1 ($-0.0006$) mà không
tạo ra sparsity (tất cả 12 trọng số đều khác 0).

% ── Bảng 2: 5-Fold CV chọn λ tối ưu ─────────────────────────────────────────
\subsection*{B.2 \ Chọn $\lambda$ Tối Ưu bằng 5-Fold Cross-Validation}

\begin{table}[H]
  \centering
  \caption{Kết quả 5-Fold Stratified CV chọn $\lambda$ (metric = F1, LR-GD, $\alpha=0.1$, 200 epochs)}
  \label{tab:cv_lambda}
  \begin{tabular}{lcc}
    \toprule
    \textbf{Loại Reg.} & \textbf{Best $\lambda$} & \textbf{CV F1 (mean $\pm$ std)} \\
    \midrule
    L1 & 0.05  & $0.5349 \pm 0.0114$ \\
    L2 & 0.001 & $0.5348 \pm 0.0114$ \\
    \bottomrule
  \end{tabular}
\end{table}

\noindent\textbf{Phân tích:}
\begin{itemize}
  \item \textbf{L1 (Lasso)} -- Best $\lambda = 0.05$: đẩy một số trọng số về 0 khi
        $\lambda$ đủ lớn, hỗ trợ feature selection; nhưng tại $\lambda=0.05$ vẫn
        chưa tạo sparsity rõ ràng trên D=12.
  \item \textbf{L2 (Ridge)} -- Best $\lambda = 0.001$: co đều các trọng số, ít
        đưa về 0; phù hợp khi nhiều feature cùng hữu ích.
  \item Cả hai cho CV F1 gần như tương đương ($\approx 0.5349$), độ lệch chuẩn
        thấp ($\pm 0.0114$) -- mô hình ổn định trên các fold.
  \item \textbf{Trade-off bias-variance:} $\lambda \to 0$ làm tăng variance (dễ
        overfit); $\lambda$ quá lớn gây underfitting. CV giúp tìm điểm cân bằng
        tối ưu.
\end{itemize}

\subsection*{B.3 \ Tác Động của Class Weighting}

\begin{table}[H]
  \centering
  \caption{Unweighted vs.\ Class-Weighted Logistic Regression (LR-GD) trên tập test}
  \label{tab:class_weight}
  \begin{tabular}{lccccc}
    \toprule
    \textbf{Cấu hình} & \textbf{Accuracy} & \textbf{Precision} & \textbf{Recall ($>$50K)} & \textbf{F1} & \textbf{AUC} \\
    \midrule
    Unweighted      & \textbf{0.823} & \textbf{0.729} & 0.425 & 0.537 & 0.845 \\
    Class-Weighted  & 0.763          & 0.505          & \textbf{0.767} & \textbf{0.609} & \textbf{0.847} \\
    \bottomrule
  \end{tabular}
\end{table}

\noindent\textbf{Phân tích:}
\begin{itemize}
  \item Class weighting tăng Recall lớp thiểu số ($>$50K) từ $0.425$ lên
        $\mathbf{0.767}$ (+34 điểm phần trăm, tương đương +80\%).
  \item Đánh đổi: Accuracy giảm 6\% và Precision giảm từ $0.729$ xuống $0.505$.
  \item F1 tăng từ $0.537$ lên $\mathbf{0.609}$ vì lợi ích từ Recall lớn hơn
        mất mát từ Precision.
  \item AUC không đổi nhiều ($0.845 \to 0.847$) -- threshold-independent metric
        ít bị ảnh hưởng bởi class weighting.
  \item \textbf{Khuyến nghị:} Nếu ứng dụng yêu cầu giảm False Negative (bỏ sót
        người thu nhập cao, ví dụ: marketing targeting), dùng \textbf{LR-Weighted}.
        Nếu cần cân bằng Precision/Recall, dùng \textbf{LR-Newton}.
\end{itemize}
